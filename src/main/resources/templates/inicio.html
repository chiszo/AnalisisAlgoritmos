Caso 4

for(int i=0; i<n; i++) {
    for(int j=i; j<n; j++) {
        if(arr[i] > arr[j]) {
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
}

Complejidad total: O(n²)

int maxPrecio(int[] precios) {
    int max = precios[0];
    for (int p : precios) {
        if (p > max)
            max = p;
    }
    return max;
}

    int busquedaBinaria(int[] arr, int id) {
    int izq = 0, der = arr.length - 1;
    while (izq <= der) {
    int mid = (izq + der) / 2;
    if (arr[mid] == id) return mid;
    if (arr[mid] < id) izq = mid + 1;
    else der = mid - 1;
}
    return -1;
    }

    boolean hayDuplicados(String[] correos) {
    for (int i = 0; i < correos.length; i++) {
        for (int j = i + 1; j < correos.length; j++) {
            if (correos[i].equals(correos[j])) return true;
        }
    }
